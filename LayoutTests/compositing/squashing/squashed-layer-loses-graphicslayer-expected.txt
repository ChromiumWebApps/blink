  (children 1
    (GraphicsLayer

      (contentsOpaque 1)
      (drawsContent 1)
      (children 1
        (GraphicsLayer
          (children 2
            (GraphicsLayer
              (position 60.00 60.00)
              (bounds 100.00 100.00)
              (contentsOpaque 1)
              (drawsContent 1)
              (backgroundColor #808080)
            )
            (GraphicsLayer
              (position 140.00 140.00)
              (bounds 260.00 260.00)
              (drawsContent 1)
            )
          )
        )
      )
    )
  )
)
